FUNCTION ZED10_EXCHANGE_RATE_TO_KRW.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IV_CURRENCY) TYPE STRING
*" REFERENCE(IV_AMOUNT) TYPE P
*" EXPORTING
*" REFERENCE(EV_KRW) TYPE P
*"----------------------------------------------------------------------

* Global data declarations

  DATA: lv_rate TYPE p DECIMALS 4.

  CASE iv_currency.
    WHEN 'USD'.
      lv_rate = '1370'.
    WHEN 'EUR'.
      lv_rate = '1480'.
    WHEN 'CNY'.
      lv_rate = '190'.
    WHEN 'JPY'.
      lv_rate = '9.2'.
  ENDCASE.

  ev_krw = iv_amount * lv_rate.




ENDFUNCTION.
